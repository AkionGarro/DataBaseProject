--returns 1(varchar2) if the tuple was created 

create or replace function  fnNewTypeClock(p_nameT in varchar2, p_descriptionT in varchar2)return varchar2 as
val varchar2(1);

begin

    packagepcd.new_TypeClock(p_nameT,p_descriptionT);
    val:='1';
    return val;
exception
    when dup_val_on_index then
     val:='0';
     dbms_output.put_line('duplicate');
     return val;
    when others then
     val:='0';
     dbms_output.put_line('duplicate');
     return val;

end fnNewTypeClock;
