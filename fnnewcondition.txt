--returns 1(varchar2) if the tuple was created 
create or replace function  fnNewCondition(p_nameC in varchar2, p_descriptionC in varchar2)return varchar2 as
val varchar2(1);

begin

    packagepcd.new_Condition(p_nameC,p_descriptionC);
    val:='1';
    return val;
exception
    when dup_val_on_index then
     val:='0';
     dbms_output.put_line('duplicate');
     return val;
    when others then
     val:='0';
     dbms_output.put_line('duplicate');
     return val;

end fnNewCondition;
