idType number(3), CONSTRAINT TYPECLOCK_pk PRIMARY KEY(idType),
nameType varchar(20) CONSTRAINT TYPECLOCK_nameType_nn not null,

createdBy VARCHAR2(20),
creationDate DATE,
modifiedBy VARCHAR2(20),
modificationDate DATE
);
