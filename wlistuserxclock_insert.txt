/*
new_WLISTUSERXCLOCK inserts a new WLISTUSERXCLOCK tuple, receives: 
                        id of APPUSER(VARCHAR2),
                        id of CLOCK(NUMBER)) 
*/

                        

create or replace procedure new_WLISTUSERXCLOCK(
                        p_idUser in VARCHAR2, 
                        p_idClock in NUMBER)

is
BEGIN

values(p_idUser,p_idClock);

END new_WLISTUSERXCLOCK;
